					set_politics = {
						ruling_party = democratic
						elections_allowed = yes
					}
					set_popularities = {
						democratic = 60
						fascism = 10
						communism = 30
						neutrality = 0
					}
				}
			}
		}
		else_if = {
			limit = {
				FIN = { has_government = fascism }
			}
			set_autonomy = { target = FIN autonomy_state = autonomy_dominion }
			hidden_effect = {
				FIN = {
					set_politics = {
						ruling_party = fascism
						elections_allowed = yes
					}
					set_popularities = {
						democratic = 10

					}
				}
			}
		}
		else = {
			set_autonomy = { target = FIN autonomy_state = autonomy_dominion }

	}
