#Monarchy

EXP_monarchism_on_the_rise = {

	prepare_for_monarchist_civil_war = {

		icon = generic_prepare_civil_war

		available = {
			neutrality > 0.1
		}

		visible = {
			NOT = { has_country_flag = preparation_for_monarchist_civil_war }
			has_idea_with_trait = staunch_monarchist
			NOT = { has_government = neutrality }
		}
		
		fire_only_once = yes
		
		cost = 50
		
		complete_effect = {
			set_country_flag = preparation_for_monarchist_civil_war
		}
		
		ai_will_do = {
			factor = 0
		}
	}
	
	expand_civil_monarchist_support = {

		icon = generic_civil_support

		visible = {
			OR = {
				has_country_flag = preparation_for_monarchist_civil_war
				has_country_flag = open_up_political_discourse_monarchist
			}
			has_idea_with_trait = staunch_monarchist
			NOT = { has_government = neutrality }
		}

		cancel_trigger = {
			hidden_trigger = {
				has_civil_war = yes
			}
		}
		
		fire_only_once = no
		
		cost = 25

		days_remove = 30

		modifier = {
			neutrality_drift = 0.02
			stability_weekly = -0.03
 			political_power_gain = -0.2
		}
		
		ai_will_do = {
			factor = 1
		}
	}
	
	army_support_for_monarchist_civil_war = {

		icon = generic_army_support

		visible = {
			has_country_flag = preparation_for_monarchist_civil_war
			has_idea_with_trait = staunch_monarchist
			NOT = { has_government = neutrality }
		}

		cancel_trigger = {
			hidden_trigger = {
				has_civil_war = yes
			}
		}
	
		fire_only_once = no
	
		cost = 50

		days_remove = 30

		modifier = {
			command_power_gain = -0.2
		}

		remove_effect = {
			if = {
				limit = {
					check_variable = { var = army_support_var value = 0.1 compare = less_than }
				}
				set_variable = { var = army_support_var value = 0.1 } 
			}
			if = {
				limit = {
					check_variable = { var = army_support_var value = 0.1 compare = greater_than_or_equals }
				}
				add_to_variable = { var = army_support_var value = 0.1 } 
			}
			set_country_flag = military_support_for_monarchist_civil_war
		}
		
		complete_effect = {
		}
		
		ai_will_do = {
			factor = 0
		}
	}

	ensure_general_loyalty_for_monarchist_civil_war = {

		icon = generic_army_support

		available = {
			has_country_flag = military_support_for_monarchist_civil_war
			command_power > 2
		}

		visible = {
			has_country_flag = preparation_for_monarchist_civil_war
			has_idea_with_trait = staunch_monarchist

				is_field_marshal = no
				NOT = { has_trait = hidden_sympathies }
			}
			NOT = { has_government = neutrality }
		}
	
		fire_only_once = yes
	
		cost = 50
		
		complete_effect = {
			country_event = { id = decisions.1 }
		}
		
		ai_will_do = {
			factor = 0
		}
	}

	ensure_marshal_loyalty_for_monarchist_civil_war = {

		icon = generic_army_support

		available = {
			has_country_flag = military_support_for_monarchist_civil_war
			command_power > 2
		}

		visible = {
			has_country_flag = preparation_for_monarchist_civil_war
			has_idea_with_trait = staunch_monarchist

				NOT = { has_trait = hidden_sympathies }
			}
			NOT = { has_government = neutrality }
			any_army_leader = {
				is_field_marshal = yes
				NOT = { has_unit_leader_flag = hidden_sympathies_marshal }
			}
		}
	
		fire_only_once = yes
	
		cost = 50
		
		complete_effect = {
			country_event = { id = decisions.2 }
		}
		
		ai_will_do = {
			factor = 0
		}
	}

	siphon_equipment_stockpiles_for_monarchist_civil_war = {

		icon = generic_prepare_civil_war

		available = {
			has_country_flag = military_support_for_monarchist_civil_war
			has_equipment = { infantry_equipment > 1000 }
			command_power > 2
		}

		visible = {
			has_country_flag = preparation_for_monarchist_civil_war
			has_country_flag = military_support_for_monarchist_civil_war
			has_idea_with_trait = staunch_monarchist
			NOT = { has_government = neutrality }
		}
	
		fire_only_once = no
	
		cost = 50
		
		complete_effect = {
			add_command_power = -2
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = -1000
			}
			set_country_flag = siphoned_equipment_stockpiles_for_civil_war
		}
		
		ai_will_do = {
			factor = 0
		}
	}

	formulate_surprise_attack_plan_for_monarchist_civil_war = {

		icon = generic_army_support

		available = {
			has_country_flag = military_support_for_monarchist_civil_war
			command_power > 2
		}

		visible = {
			has_country_flag = preparation_for_monarchist_civil_war

		complete_effect = {

		}
		
		ai_will_do = {
			factor = 0
		}
	}
	
	ignite_the_monarchist_civil_war = {

		icon = generic_ignite_civil_war

		available = {
			has_stability < 0.5
		}

		visible = {
			has_country_flag = preparation_for_monarchist_civil_war
			has_idea_with_trait = staunch_monarchist
			NOT = { has_government = neutrality }
		}
	
		fire_only_once = yes
	
		cost = 50
		
		complete_effect = {
			if = {
				limit = {
					has_government = fascism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@fascism
				}
			}
			if = {
				limit = {
					has_government = communism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@communism
				}
			}
			if = {
				limit = {
					has_government = democratic
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@democratic
				}
			}
			subtract_from_variable = {
				var = civil_war_size_var
				value = army_support_var
			}
			if = {
				limit = {
					check_variable = {
						var = civil_war_size_var
						value = 0.3
						compare = less_than
					}
				}
				set_variable = {
					var = civil_war_size_var
					value = 0.3
				}
			}
			start_civil_war = {
				ruling_party = neutrality
				ideology = ROOT
				size = civil_war_size_var
				keep_unit_leaders_trigger = {
					has_trait = hidden_sympathies
				}
			}
			clr_country_flag = preparation_for_monarchist_civil_war
			clr_country_flag = military_support_for_monarchist_civil_war
			clr_country_flag = civil_support_for_monarchist_civil_war
		}
		
		ai_will_do = {
			factor = 0
		}
	}

	rebuild_the_nation_monarchist = {

		icon = generic_construction

		visible = {
			has_country_flag = won_ideology_civil_war
		}

		fire_only_once = yes
	
		cost = 25

			political_power_gain = -1

			industrial_capacity_dockyard = 0.1

			factor = 200

			factor = 1
		}
	}
	
	discredit_government_monarchist = {

		available = {
			fascism < 0.8
			democratic < 0.8
			communism < 0.8
			neutrality < 0.8

				}

					ideology = fascism
					popularity = -0.1
				}
			}
			if = {
				limit = {
					has_government = communism
				}
				add_popularity = {
					ideology = communism
					popularity = -0.1
				}
			}
			if = {
				limit = {
					has_government = democratic
				}
				add_popularity = {
					ideology = democratic
					popularity = -0.1
				}
			}
		}
		
		ai_will_do = {
			factor = 1
		}
	}
	
	hold_the_monarchist_national_referendum = {

		icon = generic_political_discourse

		available = {
			neutrality > 0.5
		}

		visible = {
			NOT = { has_government = neutrality }
			has_idea_with_trait = staunch_monarchist
			OR = {
				neutrality > 0.6
				has_country_flag = open_up_political_discourse_monarchist
			}
		}
		
		remove_trigger = {
			has_government = neutrality
		}
		
		fire_only_once = yes
		
		cost = 100
		
		complete_effect = {
			country_event = { id = EXP_events.1 }
		}
		
		ai_will_do = {
			factor = 210
			modifier = {
				neutrality > 0.6
				factor = 10
			}
		}
	}
}

		available = {
			has_civil_war = no
			NOT = { country_exists = GER }

				limit = {
					original_tag = DDR
